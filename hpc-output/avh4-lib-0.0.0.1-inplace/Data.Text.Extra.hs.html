   19         0 -> NoSpan
   20         positive -> Span positive

   22         step (currentSpan, longest) c =
   23             if c == char
   24                 then
   25                     ( Just (1 + Maybe.fromMaybe 0 currentSpan)
   26                     , longest
   27                     )

   30                       Nothing
   31                     , -- and update the longest
