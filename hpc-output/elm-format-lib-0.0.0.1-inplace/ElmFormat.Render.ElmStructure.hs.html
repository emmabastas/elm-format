   34 forceableSpaceSepOrStack :: Bool -> Box -> [Box] -> Box
   35 forceableSpaceSepOrStack forceMultiline first rest =
   36     case
   37       ( forceMultiline, first, allSingles rest, rest )
   38     of
   39       ( False, SingleLine first', Right rest', _ ) ->
   40         line $ row $ List.intersperse space (first' : rest')
   41 
   42 
   43       _ ->
   44         stack1 (first : rest)

  112 spaceSepOrPrefix :: Box -> Box -> Box
  113 spaceSepOrPrefix op rest =
  114     case ( op, rest) of
  115         ( SingleLine op', SingleLine rest' ) ->
  116             line $ row [ op', space, rest' ]
  117 
  118         ( SingleLine op', _ ) | lineLength 0 op' < 4 ->
  119             prefix (row [ op', space ]) rest
  120 
  121         _ ->
  122             stack1 [ op, indent rest ]

  125 prefixOrIndented :: Box -> Box -> Box
  126 prefixOrIndented a b =
  127     case ( a, b ) of
  128         ( SingleLine a', SingleLine b' ) ->
  129             line $ row [ a', space, b' ]
  130 
  131         ( SingleLine a', MustBreak b' ) ->
  132             mustBreak $ row [ a', space, b' ]
  133 
  134         _ ->
  135             stack1 [ a, indent b ]

  150 equalsPair :: String -> Bool -> Box -> Box -> Box
  151 equalsPair symbol forceMultiline left right =
  152   case (forceMultiline, left, right) of
  153     ( False, SingleLine left', SingleLine right' ) ->
  154       line $ row
  155         [ left'
  156         , space
  157         , punc symbol
  158         , space
  159         , right'
  160         ]
  161 
  162     ( _, SingleLine left', MustBreak right' ) ->
  163       mustBreak $ row
  164         [ left'
  165         , space
  166         , punc symbol
  167         , space
  168         , right'
  169         ]
  170 
  171     ( _, SingleLine left', right' ) ->
  172       stack1
  173         [ line $ row [ left', space, punc symbol ]
  174         , indent right'
  175         ]
  176 
  177     ( _, left', right' ) ->
  178       stack1
  179         [ left'
  180         , indent $ line $ punc symbol
  181         , indent right'
  182         ]

  188 definition :: String -> Bool -> Box -> [Box] -> Box -> Box
  189 definition symbol forceMultiline first rest =

  208 application :: FunctionApplicationMultiline -> Box -> [Box] -> Box
  209 application forceMultiline first args =
  210   case args of
  211     [] ->
  212       first
  213 
  214     arg0 : rest ->
  215       case
  216         ( forceMultiline
  217         , first
  218         , arg0
  219         , allSingles rest
  220         )
  221       of
  222         ( FAJoinFirst JoinAll, SingleLine first', SingleLine arg0', Right rest' ) ->
  223           (first' : arg0' : rest' )
  224             |> List.intersperse space
  225             |> row
  226             |> line
  227 
  228         ( FAJoinFirst _, SingleLine first', SingleLine arg0', _) ->
  229           stack1
  230             $ line ( row [ first', space, arg0' ])
  231               : map indent rest
  232 
  233         _ ->
  234           stack1
  235             $ first : map indent (arg0 : rest)

  251 group :: Bool -> String -> String -> String -> Bool -> [Box] -> Box
  252 group innerSpaces left sep right forceMultiline children =
  253   group' innerSpaces left sep [] right forceMultiline children
  254 
  255 
  256 group' :: Bool -> String -> String -> [Box] -> String -> Bool -> [Box] -> Box
  257 group' innerSpaces left sep extraFooter right forceMultiline children =
  258   case (forceMultiline, allSingles children, allSingles extraFooter) of
  259     (_, Right [], Right efs) ->
  260       line $ row $ concat [[punc left], efs, [punc right]]
  261 
  262     (False, Right ls, Right efs) ->
  263       line $ row $ concat
  264         [ if innerSpaces then [punc left, space] else [punc left]
  265         , List.intersperse (row [punc sep, space]) (ls ++ efs)
  266         , if innerSpaces then [space, punc right] else [punc right]
  267         ]
  268 
  269     _ ->
  270       case children of
  271         [] ->
  272           -- TODO: might lose extraFooter in this case, but can that ever happen?
  273           line $ row [ punc left, punc right]
  274 
  275         (first:rest) ->
  276           stack1 $
  277             prefix (row [punc left, space]) first
  278             : map (prefix $ row [punc sep, space]) rest
  279             ++ extraFooter
  280             ++ [ line $ punc right ]

  295 extensionGroup :: Bool -> Box -> Box -> [Box] -> Box
  296 extensionGroup multiline base first rest =
  297   case
  298     ( multiline
  299     , isLine base
  300     , allSingles (first : rest)
  301     )
  302   of
  303     (False, Right base', Right fields') ->
  304       line $ row
  305         [ punc "{"
  306         , space
  307         , base'
  308         , space
  309         , punc "|"
  310         , space
  311         , row (List.intersperse (row [punc ",", space]) fields')
  312         , space
  313         , punc "}"
  314         ]
  315 
  316     _ ->
  317       stack1
  318         [ prefix (row [punc "{", space]) base
  319         , stack1
  320             ( prefix (row [punc "|", space]) first
  321             : map (prefix (row [punc ",", space])) rest)
  322             |> indent
  323         , line $ punc "}"
  324         ]
